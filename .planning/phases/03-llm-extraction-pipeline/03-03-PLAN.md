```python
import pytest
from pathlib import Path

GOLDEN_DIR = Path(__file__).parent / "golden"
FIXTURES_DIR = Path(__file__).parent / "fixtures"

@pytest.mark.parametrize("filing_fixture", list(FIXTURES_DIR.glob("*.txt")) if FIXTURES_DIR.exists() else [])
def test_extraction_regression(filing_fixture, data_regression):
    """Regression test: extraction output must match golden files.

    Run with --gen-files to regenerate golden files after intentional changes.
    Nightly CI gate — fails on any extraction delta.

    NOTE: Fixtures and golden files will be populated during bootstrap sprint.
    This test is a no-op until fixtures exist.
    """
    filing_text = filing_fixture.read_text()
    filing_type = filing_fixture.stem.split("_")[0]

    # Run extraction (requires LANGEXTRACT_API_KEY — nightly CI only)
    result = run_extraction_sync(filing_text, filing_type)

    extraction_output = [
        {
            "extraction_class": e.extraction_class,
            "extraction_text": e.extraction_text,
            "start_pos": e.char_interval.start_pos if e.char_interval else None,
            "end_pos": e.char_interval.end_pos if e.char_interval else None,
            "attributes": e.attributes,
        }
        for e in result.extractions
    ]
    data_regression.check(extraction_output)

    # Generate lx.visualize() HTML for manual inspection
    filing_stem = filing_fixture.stem
    jsonl_path = GOLDEN_DIR / f"{filing_stem}.jsonl"
    lx.io.save_annotated_documents([result], str(jsonl_path))
    viz_html_path = GOLDEN_DIR / f"{filing_stem}_viz.html"
    lx.visualize(str(jsonl_path), output_file=str(viz_html_path))
    print(f"[regression] Visualization HTML: {viz_html_path}")
```
