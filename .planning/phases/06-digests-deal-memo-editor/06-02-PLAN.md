    ```typescript
    export const memos = pgTable(
      'memos',
      {
        id: uuid('id').defaultRandom().primaryKey(),
        firmId: uuid('firm_id').references(() => firms.id).notNull(),
        dealId: uuid('deal_id').references(() => deals.id).notNull(),
        title: text('title').notNull(),
        content: jsonb('content').notNull(), // tiptap JSON
        createdBy: uuid('created_by').notNull(),
        visibility: text('visibility').notNull().default('private'), // 'private' | 'firm'
        version: integer('version').notNull().default(1), // monotonic counter for optimistic concurrency
        ...timestamps,
      },
      () => firmIsolationPolicies(),
    );

    export const memoSnapshots = pgTable(
      'memo_snapshots',
      {
        id: uuid('id').defaultRandom().primaryKey(),
        memoId: uuid('memo_id').references(() => memos.id).notNull(),
        firmId: uuid('firm_id').references(() => firms.id).notNull(),
        name: text('name').notNull(),
        content: jsonb('content').notNull(), // tiptap JSON at snapshot time
        version: integer('version').notNull(),
        createdBy: uuid('created_by').notNull(),
        createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
      },
      () => firmIsolationPolicies(),
    );
    ```
    Run `pnpm drizzle-kit push` from apps/api.

    **2. Create `apps/api/src/routes/memos.ts`:**

    Endpoints (all firm-scoped via adminDb + firmId WHERE):

    - **GET /api/memos?dealId={id}** — List memos for a deal. Filter: creator's own + visibility='firm'. Return: id, title, createdBy, visibility, version, createdAt, updatedAt (no content — too large for list).

    - **GET /api/memos/:id** — Get single memo with full content. Check: creator OR (visibility='firm' AND same firmId).

    - **POST /api/memos** — Create memo. Body: `{ dealId, title, content, visibility? }`. Sets createdBy from JWT userId, firmId from JWT. Visibility defaults to 'private' per user decision. Returns created memo with id.

    - **PATCH /api/memos/:id** — Update memo (auto-save target). Body: `{ content?, title?, visibility?, version }`. Optimistic concurrency: reject if incoming version <= stored version (return 409 Conflict). Increment version on success. Only creator can edit.

    - **DELETE /api/memos/:id** — Soft delete (set deletedAt). Only creator can delete.

    - **POST /api/memos/:id/snapshots** — Create named snapshot. Body: `{ name }`. Copies current content + version into memo_snapshots. Returns snapshot.

    - **GET /api/memos/:id/snapshots** — List snapshots for a memo. Return: id, name, version, createdBy, createdAt (no content for list).

    - **GET /api/memos/:id/snapshots/:snapshotId** — Get single snapshot with full content.

    - **POST /api/memos/:id/snapshots/:snapshotId/restore** — Restore memo content from snapshot. Sets memo.content = snapshot.content, increments version.

    Follow existing route patterns: Hono app, adminDb, explicit firmId WHERE, proper error handling (404, 403, 409).
